## 思路
递归，每个位置，对每个位置考虑，选或不选，保留状态；在返回时，清除记录
## 代码
有两种写法
第一种
```cpp
class Solution {
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
		sort(nums.begin(), nums.end());
		vector<vector<int>> ans;
		set<vector<int>> st;
		vector<int> path;
		dfs(nums, 0, st, path);
        for(auto& s : st)
        {
            ans.push_back(s);
        }
        return ans;
    }
    void dfs(vector<int>& nums, int i, set<vector<int>>& st, vector<int>& path)
    {
    	if(i == nums.size())
    	{
    		st.insert(path);
    		return;
		}
		path.push_back(nums[i]);
        dfs(nums, i + 1, st, path);
		path.pop_back();
		dfs(nums, i + 1, st, path);
	}
};
```
第二种
```cpp
class Solution {
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<vector<int>> ans;
        vector<int> path(nums.size());
        dfs(nums, 0, 0, ans, path);
        return ans;
    }
    void dfs(vector<int>& nums, int i, int size, vector<vector<int>>& ans, vector<int>& path)
    {
        if(i == nums.size())
        {
            vector<int> cur(size);
            for(int j = 0; j < size; j++)
            {
                cur[j] = path[j];
            }
            ans.push_back(cur);
            return;
        }
        int j = i + 1;
        while(j < nums.size() && nums[i] == nums[j])
        {
            j++;
        }
        dfs(nums, j, size, ans, path);
        for(; i < j; i++)
        {
            path[size++] = nums[i];
            dfs(nums, j, size, ans, path);
        }
    }
};
```
