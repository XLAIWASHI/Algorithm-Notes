## 思路
这题其实就是组合，看样例可以看出来，因为集合没有重复的，所以这就是组合，而且按任意位置排序
	那么我们可以按组的方式去看
	1. 我们先给数组排序，这样相同的元素就是一组了
	2. 接着，我们手动剪枝，手动去每一组中拿多个

## 代码
```cpp
class Solution {
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        vector<vector<int>> ans;
        vector<int> path(nums.size());
        dfs(nums, 0, 0, ans, path);
        return ans;
    }
    void dfs(vector<int>& nums, int i, int size, vector<vector<int>>& ans, vector<int>& path)
    {
        if(i == nums.size())
        {
            vector<int> cur(size);
            for(int j = 0; j < size; j++)
            {
                cur[j] = path[j];
            }
            ans.push_back(cur);
            return;
        }
        int j = i + 1; // j 是下一组第一个元素的位置
        while(j < nums.size() && nums[i] == nums[j])
        {
            j++;
        }
        dfs(nums, j, size, ans, path); // 不拿当前组的情况
        for(; i < j; i++) // 依次拿当前组的情况
        {
            path[size++] = nums[i];
            dfs(nums, j, size, ans, path);
        }
    }
};
```
