## 思路
每个元素当一次头，之后循环找没有出现过的元素

## 代码1

```cpp
class Solution {
public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> ans;
        vector<int> path;
        vector<bool> used(nums.size(), false); // 用来标记元素是否使用
        dfs(nums, 0, ans, path, used);
        return ans;
    }
    void dfs(vector<int>& nums, int i, vector<vector<int>>& ans, vector<int>& path, vector<bool>& used)
    {
        if(i == nums.size())
        {
            ans.push_back(path);
            return;
        }
        for(int j = 0; j < nums.size(); j++)
        {
            if(!used[j]) // 判断是否使用过
            {
                path.push_back(nums[j]);
                used[j] = true;
                dfs(nums, i + 1, ans, path, used);
                path.pop_back();
                used[j] = false;
            }
        }
    }
};
```

## 代码2
```cpp
class Solution {
public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> ans;
        dfs(nums, 0, ans);
        return ans;
    }
    void dfs(vector<int>& nums, int i, vector<vector<int>>& ans)
    {
        if(i == nums.size())
        {
            ans.push_back(nums);
            return;
        }
        for(int j = i; j < nums.size(); j++)
        {
            swap(nums, i, j); 
            dfs(nums, i + 1, ans);
            swap(nums, i, j);
        }
    }
    void swap(vector<int>& nums, int a, int b)
    {
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
    }
};
```
