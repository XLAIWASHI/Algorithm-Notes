## 思路
这道题的思路和上面的全排列，大差不差；将 j 换到 i 位置，但是因为这题因为是可以重复的，所以如果有相同的 j 换到 i 位置，那么所产生的排列，一定相同；所以我们就判断一下这里就可以了

## 代码
```cpp
class Solution {
public:
    vector<vector<int>> permuteUnique(vector<int>& nums) {
        vector<vector<int>> ans;
        dfs(nums, 0, ans);
        return ans;
    }
    void dfs(vector<int>& nums, int i, vector<vector<int>>& ans)
    {
        if(i == nums.size())
        {
            ans.push_back(nums);
        }
        unordered_set<int> st;
        for(int j = i; j < nums.size(); j++)
        {
            if(st.find(nums[j]) == st.end())
            {
                st.insert(nums[j]);
                swap(nums, i, j);
                dfs(nums, i + 1, ans);
                swap(nums, i, j);
            }
        }
    }
    void swap(vector<int>& nums, int a, int b)
    {
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
    }
};
```
**时间复杂度：** O(n! * n)
