## 介绍
只用递归，排序一个栈

## 代码
```cpp
#include <iostream> 
#include <algorithm>
#include <stack>
#include <limits>

int deep(std::stack<int>& sk);
int Max(std::stack<int>& sk, int deep);
int times(std::stack<int>& sk, int deep, int max_value);
void down(std::stack<int>& sk, int deep, int max_value, int times_max);

int main(void)
{
    std::stack<int> sk;
    sk.push(3);
    sk.push(2);
    sk.push(1);
    sk.push(5);
    sk.push(7);
    sk.push(4);
    sk.push(4);
    int de = deep(sk);
    int max_value;
    int max_times;
    while(de)
    {
        max_value = Max(sk, de);
        max_times = times(sk, de, max_value);
        down(sk, de, max_value, max_times);
        de -= max_times;
    }

    while(!sk.empty())
    {
        std::cout << sk.top() << std::endl;
        sk.pop();
    }
    return 0;
}

//求最大深度，不改变栈结构
int deep(std::stack<int>& sk)
{
    if(sk.empty())
    {
        return 0;
    }
    int num = sk.top();
    sk.pop();
    int ans = deep(sk) + 1;
    sk.push(num);
    return ans;
}

//获取最大值，不改变栈结构
int Max(std::stack<int>& sk, int deep)
{
    if(deep == 0)
    {
        return INT_MIN;
    }
    int num = sk.top();
    sk.pop();
    int last = Max(sk, deep - 1);
    int ans = std::max(num, last);
    sk.push(num);
    return ans;
}

//求最大值出现的次数，不改变栈结构
int times(std::stack<int>& sk, int deep, int max_value)
{
    if(deep == 0)
    {
        return 0;
    }
    int num = sk.top();
    sk.pop();
    int ans = times(sk, deep - 1, max_value) + (num == max_value ? 1 : 0);
    sk.push(num);
    return ans;
}

//最大值按照次数沉底，其他数据不变
void down(std::stack<int> &sk, int deep, int max_value, int times_max)
{
    if(deep == 0)
    {
        for(int i = 0; i < times_max; i++)
        {
            sk.push(max_value);
        }
        return;
    }
    int num = sk.top();
    sk.pop();
    down(sk, deep - 1, max_value, times_max);
    if(num != max_value)
    {
        sk.push(num);
    }
}
```
